Who said front-end code can't be modular?

index.js

const foo = require('./foo.js')

foo.bar()

foo.js

module.exports = {
  bar: function () {
    //
  }
}

Bundle your modules with webpack.

webpack.config.js

module.exports = {
  entry: './index.js',
  output: {
    path: 'bundle.js'
  }
}

page.html

<html>
  <head>
    ...
  </head>
  <body>
    ...
    <script src="/bundle.js"></script>
  </body>
</html>

It's really that simple. Get started now!
